3 消してもデフォルト値になる

contract Example {

    struct Player{
        string name;
    }

    constructor() {
        Player public bob = Player({name: "Bob"});
    }

    function deleteBob() external {
        delete bob;
    }
}

これで、 bob を消しても、 null にならない。空文字列（数字なら 0）になるだけ。id とかやるときに注意が必要。

4 Map

ちょっとでも複雑なことをしようとすると、keyvalue ペアがあると楽。 

contract Example {

    mapping (string => uint) usersAge public;

    constructor(){
        usersAge['bob'] = 35;
    }

    function getAge(string memory _user) external view returns(uint) {
        return usersAge[_user];
    }

    function setAge(string memory _user, uint memory _age) external {
        usersAge[_user] = _age;
    }

    function deleteUser(string memory _user) external {
        delete usersAge[_user]
    }

}

5 Arrays

3 種類ある。

  • Storage Arrays: ブロックチェーン状に記録されるもの

  • Memory Arrays: コントラクト内の関数内と引数として使われる配列 

    function foo(uint[] memory _ary) internal {

    uint[] memory x = _ary;
}

    である必要あり。固定長。ガスが安いから使われる。

  • Call Data Arrays: 外部データの引数として使われる配列 

    function foo(uint[] calldata _ary) external {
}

6 Internal と External functions

contract A {

    uint public x = 42;

    function externalDouble() external {
        _double(x);
    }

    function _double(uint _input) internal pure returns(uint){
        // _input is cloned, relationship of x is lost here
        _input = _input * 2;
        return _input;
    }
}

プリミティブは値渡しなので変化しません。配列とか Mapping とかは参照渡し。

7 どこにメモリを保存する?

配列や struct みたいなコピーするのにコストが高いデータは、その変数をどこに保存するか指定する必要がある。(出ないと無駄に blockchain に記録される。ガスが高いし。)

配列の種類同様、保存場所は memory, storage, calldata の 3 つ。 

contract A {

    uint[] public x = [1,2,3];

    function externalChange() external {
        _internalChange(x);
    }

    function _internalChange(uint [] storage _x) internal returns(uint[] memory){
        for(uint i in _x) {
            _x[i] = _x[i] * 2;
            return _x;
        }
    }
}

(uint [] storage _x) でブロックチェーンに記録したものを取り込んでいるので、 x は参照渡しになり、変更が反映される。

引数が storage のところを memory にすると、 _internalChange 関数内で drop されるので、値渡しすなわち、値の反映は行われない。 

function externalChange(uint[] calldata _x) public returns(uint[] memory) {
    //_x[0] = 12; // can't do this, calldata is immutable
    uint[] _y memory = _x; // pass in value (clone);

    _y[0] = 12;

    return _y;
}

8 関数

8.1 可視領域は 4 つ public, private, external, internal

public private external internal
外でも中でも コントラクト内のみ 外からのみ コントラクトとその継承

言わずもがなもっともきつい(private より)にしておいた方がセキュリティ的に良い。

8.2 制限子 pure, view, payable

pure view payable
読み書き不可 読みのみ ETH を受け取れる

8.3 modifier

modifier checkAmount(uint _amount) {
    require(msg.amount >= _amount, "Not enough ether");
    _;
}

function pay() payable external checkAmount(10 ether) {
    // do your thing
}

いろんなチェックをするのに modifier を使う、この場合、必要な eth をこのコントラクト関数に入れないと弾く仕組み。

9 コントラクトの継承

contract B {
    address payable public owner;
    constructor(){
        owner = msg.sender;
    }
}

contract A is B {

    constructor(){
        super;
    }

}

super は継承元の constructor を呼ぶ。 modifiers も子供コントラクトから読み込めます。

別ファイルのコントラクトを読み込む場合は、 

import "./B.sol";

とすれば良い。

10 外部コントラクトの利用

*継承ではない 

contract Callable {
    uint public number = 20;
    function getNumber() external returns(uint){
        return number;
    }
}


import "./Callable.sol";

contract Caller{

    address public callableAddress;

    function setCallableAddress(address _address) external {
        callableAddress = _address;
    }

    function getCallableNumber() external {
        Callable callable = Callable(callableAddress);
        return callable.getNumber();
    }
}

Collable.sol を先にデプロイしておく必要あり。それで setCallableAddress にアドレスを入れてあげて、それを使っていく。

11 ライブラリを使う

ライブラリを使うのは、外部コントラクトを使ったり、継承とはちょっと違う。ライブラリは計算のコンテキストが呼び出した側なのに対して、他の二つの例は呼び出された環境。

github.com でおもむろにファイルを指定して 

import "https://....foo.sol";

contract A{
    using Foo for uint;

    uint public x = 10;

    function addx(uint _z) external view returns(uint) {
        return x.add(_z);
    }
}

using XXX for uint; の箇所はちょっとインターフェースだったりダックタイピング的なシンタックス。 

function add(uint _x, uint _y) internal pure returns(uint) {
    return _x + _y;
}

みたいなことが起きている想定で、第一引数が self 見たくなる。
