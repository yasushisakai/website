use sha2::{Digest, Sha256};
use std::fmt;

fn main() {
    let bits = 386_673_244u32;
    let target_bytes = bits.to_be_bytes();

    // create 32byte mutable array.
    let mut target: [u8; 32] = [0; 32];
    target[32 - (target_bytes[0] as usize)] = target_bytes[1];
    target[32 - (target_bytes[0] as usize) + 1] = target_bytes[2];
    target[32 - (target_bytes[0] as usize) + 2] = target_bytes[3];

    println!("{:?}", target.to_vec());

    let raw = Sha256::digest(b"abc");

    println!("{:x}", &raw);

    let h: Sha256Hash = raw.into();

    println!("{:x}", h);

    // println!("{}", h > target);
}

struct Sha256Hash([u8; 32]);

// type Sha256Hash = [u8;32];

impl fmt::LowerHex for Sha256Hash {
    fn fmt(&self, _f: &mut fmt::Formatter<'_>) -> fmt::Result {
        let val = self.0;
        let result = val
            .iter()
            .map(|n| format!("{:02x}", n))
            .fold(String::new(), |acc, n| format!("{}{}", acc, n));
        println!("{}", result);
        Ok(())
    }
}

impl From<[u8;32]> for Sha256Hash {
    fn from(num: [u8;32]) -> Self {
        Self(num)
    }
}

use sha2::{Digest, Sha256};

fn main() {
    let hash = Sha256::digest(b"abc");
    println!("{:?}", hash);
    let hex_digest = format!("{:x}", hash); // 64 chars

    let mut bytes = [0; 32];

    for i in 0..32 {
        let a = hex_digest.chars().nth(i * 2).unwrap().to_digit(16).unwrap() as u8;
        let b = hex_digest
            .chars()
            .nth(i * 2 + 1)
            .unwrap()
            .to_digit(16)
            .unwrap() as u8;
        let c = a * 16 + b;
        bytes[i] = c;
    }

    println!("{:?}", bytes);
}
